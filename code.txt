<script>
(()=>{
const N=8;
const $=id=>document.getElementById(id);
const board=$('board');
const rnd=n=>Math.random()*n|0;

// piece shapes (relative cells)
const SHAPES=[
  [[0,0],[0,1]],                 // domino
  [[0,0],[0,1],[0,2]],           // I3
  [[0,0],[0,1],[1,0]],           // corner L
  [[0,0],[0,1],[1,0],[1,1]],     // square
  [[0,0],[0,1],[0,2],[1,0]],     // J
  [[0,0],[0,1],[1,1],[1,2]],     // S
  [[0,0],[0,1],[0,2],[1,1]],     // T
  [[0,0]],                        // single (rare)
];
function randShape(){
  const r=Math.random();
  const pool=r<.12?[[[0,0]]]:SHAPES.slice(0,7);
  return pool[rnd(pool.length)].map(p=>[...p]);
}
function rotate(cells){ // 90deg
  const r=cells.map(([y,x])=>[x,-y]);
  const miny=Math.min(...r.map(p=>p[0])),minx=Math.min(...r.map(p=>p[1]));
  return r.map(([y,x])=>[y-miny,x-minx]);
}

let S=null;
function start(){
  // grid: 0 grass,1 wall,2 fire,3 burnt
  const grid=Array.from({length:N},()=>Array(N).fill(0));
  // 2 fire seeds near edges/corners
  const seeds=[[0,rnd(3)],[N-1,N-1-rnd(3)]];
  seeds.forEach(([r,c])=>grid[r][c]=2);
  // houses: 3-4 placed away from seeds
  const houses=[];
  let tries=0;
  while(houses.length<4&&tries++<200){
    const r=2+rnd(N-4),c=1+rnd(N-2);
    if(grid[r][c]!==0)continue;
    if(houses.some(h=>Math.abs(h[0]-r)+Math.abs(h[1]-c)<2))continue;
    if(seeds.some(s=>Math.abs(s[0]-r)+Math.abs(s[1]-c)<3))continue;
    houses.push([r,c]);
  }
  S={grid,houses,turn:1,tray:[randShape(),randShape(),randShape(),randShape(),randShape(),randShape()],
     sel:0,over:false,ghost:null};
  buildBoard();renderTray();render();
}

function buildBoard(){
  board.innerHTML='';
  for(let r=0;r<N;r++)for(let c=0;c<N;c++){
    const d=document.createElement('div');
    d.className='c';d.dataset.r=r;d.dataset.c=c;
    d.addEventListener('pointerenter',()=>hover(r,c));
    d.addEventListener('pointerdown',()=>{hover(r,c);placeSel(r,c)});
    board.appendChild(d);
  }
}
const cellAt=(r,c)=>board.children[r*N+c];
const isHouse=(r,c)=>S.houses.some(h=>h[0]===r&&h[1]===c);

function cellsFor(shape,r,c){return shape.map(([dy,dx])=>[r+dy,c+dx])}
function canPlace(shape,r,c){
  return cellsFor(shape,r,c).every(([y,x])=>
    y>=0&&y<N&&x>=0&&x<N&&S.grid[y][x]===0&&!isHouse(y,x));
}

function hover(r,c){
  if(S.over)return;
  S.ghost=[S.tray[S.sel],r,c];
  render();
}
function placeSel(r,c){
  if(S.over)return;
  const shape=S.tray[S.sel];
  if(!canPlace(shape,r,c))return;
  cellsFor(shape,r,c).forEach(([y,x])=>S.grid[y][x]=1);
  // consume piece, shift queue, draw new
  S.tray.splice(S.sel,1);
  S.tray.push(randShape());
  S.sel=0;S.ghost=null;
  spreadFire();
}

function spreadFire(){
  const add=[];
  for(let r=0;r<N;r++)for(let c=0;c<N;c++){
    if(S.grid[r][c]!==2)continue;
    [[1,0],[-1,0],[0,1],[0,-1]].forEach(([dy,dx])=>{
      const y=r+dy,x=c+dx;
      if(y>=0&&y<N&&x>=0&&x<N&&S.grid[y][x]===0)add.push([y,x]);
    });
  }
  add.forEach(([y,x])=>S.grid[y][x]=2);
  // house hit?
  const lost=S.houses.find(h=>S.grid[h[0]][h[1]]===2);
  if(lost){S.over='lose';render();setTimeout(()=>finish(false),500);return}
  // fire fully contained?
  if(add.length===0&&!anyFireCanGrow()){S.over='win';render();setTimeout(()=>finish(true),500);return}
  S.turn++;
  render();renderTray();
}
function anyFireCanGrow(){
  for(let r=0;r<N;r++)for(let c=0;c<N;c++){
    if(S.grid[r][c]!==2)continue;
    for(const[dy,dx]of[[1,0],[-1,0],[0,1],[0,-1]]){
      const y=r+dy,x=c+dx;
      if(y>=0&&y<N&&x>=0&&x<N&&S.grid[y][x]===0)return true;
    }
  }
  return false;
}
// cells that will catch fire on the NEXT fire step (preview)
function nextBurn(){
  const set=new Set();
  for(let r=0;r<N;r++)for(let c=0;c<N;c++){
    if(S.grid[r][c]!==2)continue;
    [[1,0],[-1,0],[0,1],[0,-1]].forEach(([dy,dx])=>{
      const y=r+dy,x=c+dx;
      if(y>=0&&y<N&&x>=0&&x<N&&S.grid[y][x]===0)set.add(y*N+x);
    });
  }
  return set;
}

function render(){
  const burn=nextBurn();
  const ghostCells=new Set();
  let ghostOk=true;
  if(S.ghost){
    const[sh,r,c]=S.ghost;
    ghostOk=canPlace(sh,r,c);
    cellsFor(sh,r,c).forEach(([y,x])=>{if(y>=0&&y<N&&x>=0&&x<N)ghostCells.add(y*N+x)});
  }
  for(let r=0;r<N;r++)for(let c=0;c<N;c++){
    const d=cellAt(r,c),v=S.grid[r][c],k=r*N+c;
    let cls='c';
    if(v===1)cls+=' wall';
    else if(v===2)cls+=' fire';
    else if(v===3)cls+=' burnt';
    if(isHouse(r,c)&&v===2)cls+=' house-lost';
    if(!S.over&&v===0&&burn.has(k)&&!ghostCells.has(k))cls+=' willburn';
    if(ghostCells.has(k))cls+=ghostOk?' ghost':' ghost-bad';
    d.className=cls;
    // icon
    let ico='';
    if(isHouse(r,c))ico=S.grid[r][c]===2?'🔥':'🏠';
    else if(v===2)ico='🔥';
    d.innerHTML=ico?`<span class="ico">${ico}</span>`:'';
  }
  $('houses').textContent=S.houses.filter(h=>S.grid[h[0]][h[1]]!==2).length+'/'+S.houses.length;
  $('turn').textContent=S.turn;
  const warn=$('warn');
  const threat=S.houses.some(h=>{
    return [[1,0],[-1,0],[0,1],[0,-1]].some(([dy,dx])=>{
      const y=h[0]+dy,x=h[1]+dx;
      return y>=0&&y<N&&x>=0&&x<N&&S.grid[y][x]===2;
    });
  });
  warn.textContent=threat?'⚠ Огонь у дома! Защити его этим ходом':'';
}

function renderTray(){
  const box=$('pieces');box.innerHTML='';
  S.tray.forEach((sh,i)=>{
    if(i===3){const sep=document.createElement('div');sep.className='sep';box.appendChild(sep)}
    const active=i<3;
    const p=document.createElement('div');
    p.className='piece'+(active?'':' next')+(i===S.sel?' sel':'');
    const maxY=Math.max(...sh.map(c=>c[0])),maxX=Math.max(...sh.map(c=>c[1]));
    const mini=document.createElement('div');mini.className='mini';
    mini.style.gridTemplateColumns=`repeat(${maxX+1},11px)`;
    const filled=new Set(sh.map(([y,x])=>y*10+x));
    for(let y=0;y<=maxY;y++)for(let x=0;x<=maxX;x++){
      const cell=document.createElement('i');
      if(!filled.has(y*10+x))cell.className='e';
      mini.appendChild(cell);
    }
    p.appendChild(mini);
    if(active)p.addEventListener('pointerdown',e=>{e.stopPropagation();S.sel=i;S.ghost=null;renderTray();render()});
    box.appendChild(p);
  });
  $('pass').textContent='в очереди: 3';
}

$('rotateBtn').onclick=()=>{if(S.over)return;S.tray[S.sel]=rotate(S.tray[S.sel]);S.ghost=null;renderTray();render()};
$('restartBtn').onclick=()=>start();
board.addEventListener('pointerleave',()=>{S.ghost=null;render()});

function finish(win){
  const saved=S.houses.filter(h=>S.grid[h[0]][h[1]]!==2).length;
  if(win){
    const score=saved*100+Math.max(0,(40-S.turn))*10;
    show(`<h2 class="ok">🛡 Огонь заперт!</h2>
      <div class="big ok">${score}</div>
      <p>Спасено домов: <b>${saved}/${S.houses.length}</b>. Заперто за <b>${S.turn}</b> ходов — чем быстрее, тем больше очков.</p>
      <button class="btn btn-go" id="go">Ещё пожар</button>`);
  }else{
    show(`<h2 class="bad">🔥 Дом сгорел</h2>
      <div class="big bad">${saved}/${S.houses.length}</div>
      <p>Огонь прорвался к дому. Видел, какие клетки загорятся, — где надо было ставить стенку?</p>
      <button class="btn btn-go" id="go">Ещё раз</button>`);
  }
  $('go').onclick=()=>{hide();start()};
}
function showIntro(){
  show(`<div class="subtitle">прототип</div><div class="title">БРАНДМАУЭР</div>
    <div class="rules">
      <div><span class="i">🔥</span><span>Огонь расползается каждый ход — на 4 соседние клетки. По стенам и выгоревшему не идёт.</span></div>
      <div><span class="i">🧱</span><span>Ставь стенки из фигур (3 активные + 3 в очереди). Поворот — кнопкой. Сначала ты, потом огонь.</span></div>
      <div><span class="i">🏠</span><span>Запри огонь, пока он не дошёл до домов. Пунктир — куда перекинется следующим ходом.</span></div>
      <div><span class="i">⏱</span><span>Чем быстрее заперт огонь — тем больше очков. Иногда выгоднее отдать угол и спасти главное.</span></div>
    </div>
    <button class="btn btn-go" id="go">Тушить</button>`);
  $('go').onclick=()=>hide();
}
const ovl=$('ovl');
function show(html){ovl.innerHTML=`<div class="ovl"><div class="card">${html}</div></div>`}
function hide(){ovl.innerHTML=''}

start();showIntro();
})();
</script>
